\begin{frame}
\frametitle{O Problema}
\framesubtitle{Formalmente}
	\begin{description}
	 \item [Instância] Um grafo G = (T,P) direcionado acíclico sobre um conjunto de
	 tarefas T, o tempo de execução $t_i$ de
	cada tarefa $i \in T$, e um número $m$ de estações de trabalho.

	 \item [Solução] Uma atribuição $s : T \rightarrow [m]$ das tarefas às
	 estações que satisfaz as restrições de
	precedência, i.e. para cada $tu \in P$, $s(t) \leq s(u)$.

	 \item [Objetivo] Minimizar o tempo de ciclo  $max_{i\in[m]}\sum_{j\in T\mid s(t)=i}t_j$.
	\end{description}
\end{frame}



\begin{frame}
\frametitle{Formulação Inteira}

	Seja $T$ o conjunto de tarefas e $M$ o conjunto de máquinas. $x_{ij}$ é a variável que indica se a tarefa $i$ foi atribuída à máquina $j$.

	\begin{center}
	\begin{tabular}{ r l l }
	  \textbf{min} & T &\\
	  \textbf{s.a} & $ \sum_{i=1}^{n} t_i \cdot x_{ij} \leq T $ & $\forall j \in M$\\
	& $ \sum_{j=1}^{m} x_{ij} = 1$ & $\forall i \in T$ \\
	& $ \sum_{j=1}^{m} j \cdot x_{uj} \leq  \sum_{j=1}^{m} j \cdot x_{vj} $ & $\forall (u,v) \mid PRECEDE[u,v] = 1 $\\
	& $ x_{ij} \in \mathbb{B} $ & $\forall i \in T$, $\forall j \in M$ \\
	\end{tabular}
	\end{center}

\end{frame}

\begin{frame}[fragile]
\frametitle{Implementação GLPK}

	\begin{lstlisting}
param M;
param N;
param costs{1..N} >= 0;
param prec{1..N, 1..N} >=0;

var cicletime, integer;
var delegate{1..N,1..M} >= 0, binary;


minimize minCicleTime:
    cicletime;

subject to cicleTimeLimit {j in 1..M}:
    (sum {i in 1..N} costs[i]*delegate[i,j]) <= cicletime;

subject to delegateAll {i in 1..N}:
    (sum {j in 1..M} delegate[i,j]) = 1;

subject to respectPrecedences {u in 1..N, v in 1..N}:
    if prec[u,v] == 1 then
        (sum {j in 1..M} j*delegate[u,j]) <= (sum {j in 1..M} j*delegate[v,j]);
	\end{lstlisting}
\end{frame}

\begin{frame}
\frametitle{Modelagem GRASP}
\framesubtitle{Cálculo da Vizinhança}

Sendo m* a máquina com maior carga...

\begin{description}
	\item[Simples] ...retira \textbf{uma} tarefa de m* e coloca em uma outra máquina.
	\item[Melhorada] ...retira \textbf{cada} tarefa de m* e coloca em uma outra máquina.
\end{description}

Mais de uma máquina m* ... ?

\end{frame}


\begin{frame}
\frametitle{Modelagem GRASP - Primeira Abordagem}

\begin{enumerate}
	
	\item Pega tarefas uma a uma da lista de tarefas
	\item Adiciona a tarefa na máquina que minimiza o ciclo máximo e respeita as restrições
	\item Repete para cada tarefa
	
\end{enumerate}

\end{frame}
